make distclean
./configure \
    --prefix="../ffmpeg_build" \
    --target-os=mingw32 \
    --arch=x86_64 \
    --enable-static \
    --disable-shared \
    --disable-all \
    --disable-everything \
    --enable-small \
    --enable-avformat \
    --enable-avcodec \
    --enable-avutil \
    --enable-protocol=file \
    --enable-demuxer=mp3,mov,m4a,flac,ogg \
    --enable-parser=mpegaudio,flac,aac \
    --disable-network \
    --disable-debug \
    --disable-doc

make -j$(nproc)
make install
